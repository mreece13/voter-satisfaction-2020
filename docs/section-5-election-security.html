Q5.1_1_data <- df %>%
  drop_na(Q5.1_familyQuestionsFrequency) %>%

  rename(response = Q5.1_familyQuestionsFrequency) %>%

  mutate(response = fct_relevel(response, "All or almost all of the time", "Most of the time", "Sometimes", "Rarely")) %>%
